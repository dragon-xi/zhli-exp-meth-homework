    TH1F *sum5=new TH1F("sum5","sum 1~5",4000,0,4000);
    TH1F *sum10=new TH1F("sum10","sum 1~10",4000,0,4000);

        double rx=0;

            if(j==9) sum10->Fill(rx*10);
            if(j==19) sum20->Fill(rx*5);
            if(j==99) sum100->Fill(rx*1);
        }
    }
